create database funeraria;

create table produtos (id int(3) not null auto_increment primary key, nome varchar(200) not null, preco decimal(10.2) not null) engine innodb;

create table eventos (id int(9) not null auto_increment primary key, nome varchar(80) not null, cpf char(14) not null, telefone1 int(20) not null, telefone2 int(20) not null, email varchar(90) not null, produto_id varchar(200) not null,
 receita int(7)) engine innodb;

create table despesas (id int(3) not null auto_increment primary key, nome varchar(150) not null, km int(7) not null, despesa int(7) not null, obs varchar(100), evento_id int(7)) engine innodb;
