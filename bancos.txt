create database funeraria;
create table produtos (id int(3) not null auto_increment primary key, nome varchar(200) not null, preco decimal(10.2) not null) engine innodb;

create table eventos (id int(9) not null auto_increment primary key, nome varchar(80) not null, cpf char(14) not null, telefone1 int(20) not null, telefone2 int(20) not null, email varchar(90) not null, produto_id int(3) not null, foreign key (produto_id) references produtos(id) on update cascade on delete cascade) engine innodb;
